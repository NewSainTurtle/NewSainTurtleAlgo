import java.io.*;

public class Main {
  static int[] dx = { -1, 0, 1, 0 }; // â†‘ â†’ â†“ â†

    StringTokenizer st = new StringTokenizer(br.readLine(), " ");
    int R = Integer.parseInt(st.nextToken());
    int C = Integer.parseInt(st.nextToken());

    char[][] bigBombMap = new char[R][C];

            miniBombMap[i][j] = '.';
            for (int k = 0; k < 4; k++) {
              int nx = i + dx[k];
              int ny = j + dy[k];

              for (int k = 0; k < 4; k++) {
                int nx = i + dx[k];
                int ny = j + dy[k];
